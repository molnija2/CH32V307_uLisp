
(defun copy-file (source dest)
 (if (and (stringp source) (stringp dest))
  (with-sd-card (wrstr dest 2)
   (with-sd-card (rdstr source 0)
     (let ((c 0))
      (loop
         (setq c (read-byte rdstr))
          (if (equal c nil) (return))
          (write-byte c wrstr)
       ))) t)
  nil
 )
)
